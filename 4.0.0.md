public class LineOffsetDetector {
    
    // æ£€æµ‹å•ä¸ª TopicLine
    public OffsetInfo detectOffset(Project project, TopicLine line) {
        // 1. æ£€æŸ¥æ–‡ä»¶
        if (line.file() == null || !line.file().isValid()) {
            return OffsetInfo.fileMissing(line);
        }
        
        // 2. æ£€æŸ¥ Bookmark
        String uid = line.getBookmarkUid();
        if (uid == null) {
            return OffsetInfo.bookmarkMissing(line, "è€æ•°æ®æ²¡æœ‰ UID");
        }
        
        Bookmark bookmark = BookmarkUtils.machBookmark(line, project);
        if (bookmark == null) {
            return OffsetInfo.bookmarkMissing(line, "Bookmark å·²åˆ é™¤");
        }
        
        // 3. æ¯”å¯¹è¡Œå·
        int currentLine = line.line();
        int bookmarkLine = getBookmarkLine(bookmark);
        
        if (currentLine == bookmarkLine) {
            return OffsetInfo.synced(line);
        } else {
            return OffsetInfo.offset(line, currentLine, bookmarkLine);
        }
    }
    
    // æ‰¹é‡æ£€æµ‹
    public Map<TopicLine, OffsetInfo> detectAll(Project project) {
        CodeReadingNoteService service = CodeReadingNoteService.getInstance(project);
        List<TopicLine> allLines = service.getTopicList().getTopics().stream()
            .flatMap(topic -> topic.getLines().stream())
            .collect(Collectors.toList());
            
        return allLines.stream()
            .collect(Collectors.toMap(
                line -> line,
                line -> detectOffset(project, line)
            ));
    }
}

public class OffsetInfo {
    private TopicLine topicLine;
    private OffsetStatus status;
    private int originalLine;
    private int newLine;
    private int offset;  // newLine - originalLine
    private String reason;
    
    public static OffsetInfo synced(TopicLine line) {

        return new OffsetInfo(line, OffsetStatus.OFFSET, original, newLine, newLine - original, null);

    public FixResult fixLine(Project project, TopicLine line) {

        List<TopicLine> allLines = service.getTopicList().getTopics().stream()

            .collect(Collectors.toList());
            
        return fixLines(project, allLines);
    }

    private FixResult fixLines(Project project, List<TopicLine> lines) {
        Map<String, Bookmark> bookmarkMap = BookmarkUtils.getStringBookmarkMap(project);
        
        int total = 0;
        int fixed = 0;
        int failed = 0;
        
        for (TopicLine line : lines) {
            OffsetInfo info = detector.detectOffset(project, line);
            
            if (info.getStatus() == OffsetStatus.OFFSET) {
                total++;
                try {
                    line.modifyLine(info.getNewLine());
                    fixed++;
                } catch (Exception e) {
                    failed++;
                }
            }
        }
        
        // é€šçŸ¥ UI åˆ·æ–°
        CodeReadingNoteService.getInstance(project).getTopicList().notifyListeners();
        
        return new FixResult(fixed, total, failed);
    }
    
    // è‡ªåŠ¨ä¿®å¤ï¼ˆå¼‚æ­¥ï¼‰
    public CompletableFuture<FixResult> autoFixAsync(Project project, FixTrigger trigger) {
        return CompletableFuture.supplyAsync(() -> {
            // æ ¹æ®è§¦å‘å™¨å’Œé…ç½®å†³å®šæ˜¯å¦ä¿®å¤
            AutoFixSettings settings = AutoFixSettings.getInstance();

        }, ApplicationManager.getApplication()::executeOnPooledThread);
    }
